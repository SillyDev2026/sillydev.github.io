Pipeline Workflow Engine

Create reusable data pipelines with step chains, parallel execution, branch routing, filters, repeat loops, hooks, and error handling. Perfect for game systems, processing flows, and structured logic.

Overview

Main features included in this class

Step Chains

Run multiple operations in sequence with readable pipeline syntax.

Parallel Tasks

Execute multiple functions and merge results together.

Branch Routing

Send data into different pipelines based on conditions.

Filters & Loops

Stop flows, validate input, or repeat stages until complete.

Core Methods

Build and run custom pipelines

  • Pipeline.new() — create a new pipeline
  • :step() — add sequential stage
  • :parallel() — run multiple stages together
  • :branch() — route to child pipelines
  • :filter() — halt when condition fails
  • :repeatStage() — repeat until condition ends
  • :merge() — merge parallel results
  • :catch() — catch runtime errors
  • :finally() — always run cleanup callback
  • :hooksFn() — before/after stage hooks
  • :run(input) — execute the pipeline

Quick Example

Simple chained pipeline


local Pipeline = require(path.to.Pipeline)

local pipe = Pipeline.new()

pipe
:step(function(v)
    return v + 5
end)

:step(function(v)
    return v * 2
end)

local result = pipe:run(10)
print(result) -- 30

--!native
--!optimize 2

export type Context = { [string]: any }

export type StageFn = (input: I, ctx: Context) -> O
export type BranchConditionFn = (input: I, ctx: Context) -> string

export type Stage = {
	type: "step" | "parallel" | "branch" | "filter" | "repeat",
	fn: StageFn?,
	fns: { StageFn }?,
	stageFn: StageFn?,
	conditionFn: BranchConditionFn?,
	branches: { [string]: Pipeline }?,
	label: string?,
}

export type Pipeline = {
	stages: { Stage },
	context: Context,
	hooks: {
		beforeStage: ((label: string, value: any, ctx: Context) -> ())?,
		afterStage: ((label: string, value: any, ctx: Context) -> ())?,
	},
	mergeFn: ((results: { any }, ctx: Context) -> any)?,
	catchFn: ((err: any, ctx: Context) -> any)?,
	finallyFn: ((result: any, ctx: Context) -> ())?,

	step: (self: Pipeline, fn: StageFn, label: string?) -> Pipeline,
	parallel: (self: Pipeline, fns: { StageFn }, label: string?) -> Pipeline,
	branch: (self: Pipeline, condFn: BranchConditionFn, branches: { [string]: Pipeline }, label: string?) -> Pipeline,
	filter: (self: Pipeline, fn: StageFn, label: string?) -> Pipeline,
	repeatStage: (self: Pipeline, stageFn: StageFn, condFn: BranchConditionFn, label: string?) -> Pipeline,

	merge: (self: Pipeline, fn: (results: { any }, ctx: Context) -> O) -> Pipeline,
	catch: (self: Pipeline, fn: (err: any, ctx: Context) -> O) -> Pipeline,
	finally: (self: Pipeline, fn: (result: O, ctx: Context) -> ()) -> Pipeline,
	hooksFn: (self: Pipeline, hooks: { beforeStage: ((string, any, Context) -> ())?, afterStage: ((string, any, Context) -> ())? }) -> Pipeline,

	run: (self: Pipeline, input: I) -> (O, Context),
}

local Pipeline = {}
Pipeline.__index = Pipeline

local HALT = {}

function Pipeline.new(): Pipeline
	return setmetatable({
		stages = {},
		context = {},
		hooks = { beforeStage = nil, afterStage = nil },
		mergeFn = nil,
		catchFn = nil,
		finallyFn = nil,
	}, Pipeline) :: Pipeline
end

function Pipeline.step(self: Pipeline, fn: StageFn, label: string?): Pipeline
	table.insert(self.stages, {
		type = "step",
		fn = fn,
		label = label,
	})
	return self
end

function Pipeline.parallel(self: Pipeline, fns: { StageFn }, label: string?): Pipeline
	table.insert(self.stages, {
		type = "parallel",
		fns = fns,
		label = label,
	})
	return self
end

function Pipeline.branch(self: Pipeline, condFn: BranchConditionFn, branches: { [string]: Pipeline }, label: string?): Pipeline
	table.insert(self.stages, {
		type = "branch",
		conditionFn = condFn,
		branches = branches,
		label = label,
	})
	return self
end

function Pipeline.filter(self: Pipeline, fn: StageFn, label: string?): Pipeline
	table.insert(self.stages, {
		type = "filter",
		fn = fn,
		label = label,
	})
	return self
end

function Pipeline.repeatStage(self: Pipeline, stageFn: StageFn, condFn: BranchConditionFn, label: string?): Pipeline
	table.insert(self.stages, {
		type = "repeat",
		stageFn = stageFn,
		conditionFn = condFn,
		label = label,
	})
	return self
end

function Pipeline.merge(self: Pipeline, fn: (results: { any }, ctx: Context) -> O): Pipeline
	self.mergeFn = fn
	return self
end

function Pipeline.hooksFn(self: Pipeline, hooks: { beforeStage: ((string, any, Context) -> ())?, afterStage: ((string, any, Context) -> ())? }): Pipeline
	self.hooks = hooks
	return self
end

function Pipeline.catch(self: Pipeline, fn: (err: any, ctx: Context) -> O): Pipeline
	self.catchFn = fn
	return self
end

function Pipeline.finally(self: Pipeline, fn: (result: O, ctx: Context) -> ()): Pipeline
	self.finallyFn = fn
	return self
end

function Pipeline.run(self: Pipeline, input: I): (O, Context)
	local result: any = input
	local ctx = self.context

	local function runStage(stage: Stage)
		if self.hooks.beforeStage then
			self.hooks.beforeStage(stage.label or stage.type, result, ctx)
		end

		local val: any

		if stage.type == "step" and stage.fn then
			val = stage.fn(result, ctx)

		elseif stage.type == "parallel" and stage.fns then
			local results = {}
			for i, fn in ipairs(stage.fns) do
				results[i] = fn(result, ctx)
			end
			val = self.mergeFn and self.mergeFn(results, ctx) or results

		elseif stage.type == "branch" and stage.conditionFn and stage.branches then
			local key = stage.conditionFn(result, ctx)
			local branchPipeline = stage.branches[key] or stage.branches["other"]

			if branchPipeline then
				val = branchPipeline:run(result)
			else
				val = result
			end

		elseif stage.type == "filter" and stage.fn then
			val = stage.fn(result, ctx) and result or HALT

		elseif stage.type == "repeat" and stage.stageFn and stage.conditionFn then
			local loopResult = result
			while stage.conditionFn(loopResult, ctx) do
				loopResult = stage.stageFn(loopResult, ctx)
			end
			val = loopResult

		else
			val = result
		end

		if self.hooks.afterStage then
			self.hooks.afterStage(stage.label or stage.type, val, ctx)
		end

		return val
	end

	for _, stage in ipairs(self.stages) do
		local ok, val = pcall(runStage, stage)

		if ok then
			if val == HALT then break end
			result = val
		else
			if self.catchFn then
				return self.catchFn(val, ctx), ctx
			else
				error(val)
			end
		end
	end

	if self.finallyFn then
		self.finallyFn(result, ctx)
	end

	return result :: O, ctx
end

return Pipeline
