Roblox Module

Promise Documentation

A Lua (Luau) async system for Roblox featuring chaining, retries, event handling, and async utilities.

Installation

local Promise = require(path.to.Promise)

Basic Usage

local Promise = require(PromiseModule)

Promise.new(function(resolve, reject)
    task.wait(1)
    resolve("Hello World")
end):andThen(function(result)
    print(result)
end)

Chaining

Promise.new(function(resolve)
    resolve(10)
end)
:andThen(function(value)
    return value * 2
end)
:andThen(function(value)
    print(value) -- 20
end)

Error Handling

Promise.new(function(_, reject)
    reject("Something went wrong")
end)
:catch(function(err)
    warn(err)
end)

Finally

Promise.new(function(resolve)
    resolve("Done")
end)
:finally(function()
    print("Always runs")
end)

Advanced Examples

Promise.all

Promise.all({
    Promise.resolve(1),
    Promise.resolve(2),
    Promise.resolve(3)
}):andThen(function(results)
    print(results[1], results[2], results[3])
end)

Retry

Promise.retry(function(resolve, reject)
    if math.random() > 0.5 then
        resolve("Success")
    else
        reject("Failed")
    end
end, 3)

Delay

Promise.delay(2):andThen(function()
    print("2 seconds passed")
end)

Events

local Players = game:GetService("Players")

Promise.fromEvent(Players.PlayerAdded)
:andThen(function(player)
    print(player.Name .. " joined")
end)

Timeout

Promise.timeOut(
    Promise.delay(10),
    3
):catch(function(err)
    warn(err)
end)

HTTP Example

Promise.sendToDiscord(
    "WEBHOOK_URL",
    { content = "Hello from Promise!" }
)
:andThen(print)
:catch(warn)
