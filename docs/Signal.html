Signal Module

Fast Event System

Lightweight Luau signal implementation with Connect, Once, Wait, Fire, cleanup utilities, and native optimization flags.

Get Module Back to Modules

Installation


local Signal = require(path.to.Signal)

Basic Usage


local Signal = require(SignalModule)

local Changed = Signal.new()

Changed:Connect(function(value)
    print("Changed:", value)
end)

Changed:Fire(25)

Once Listener


local Hit = Signal.new()

Hit:Once(function()
    print("Only runs once")
end)

Hit:Fire()
Hit:Fire()

Wait()


task.spawn(function()
    local value = Changed:Wait()
    print("Received:", value)
end)

Changed:Fire("Hello")

Disconnect


local SignalObject = Signal.new()

local connection = SignalObject:Connect(function()
    print("Connected")
end)

connection:Disconnect()

Destroy


local Event = Signal.new()

Event:Destroy()
