DataStore Module


--!optimize 2
--!native

local DataStoreService = game:GetService("DataStoreService")
local Players = game:GetService("Players")

local PlayersData = require(script.Parent.PlayersData)

export type Config = {
	name: string,
	scope: string?,
	template: PlayersData.Data,
	AutoSaveInterval: number?,
	maxRetries: number?,
	Debug: boolean?,
}

export type Profile = {
	userId: number,
	key: string,
	Data: PlayersData.Data,
	loaded: boolean,
	dirty: boolean,
	sessionlock: boolean,
	lastsave: number,
}

local DEFAULT_AUTOSAVE = 30
local DEFAULT_RETRIES = 5

local DataStoreClass = {}
DataStoreClass.__index = DataStoreClass

function DataStoreClass.new(config: Config)
	assert(config.name, "Missing datastore name")
	assert(config.template, "Missing template")

	local self = setmetatable({}, DataStoreClass)

	self.Store = DataStoreService:GetDataStore(
		config.name,
		config.scope or "global"
	)

	self.template = config.template
	self.autosave = config.AutoSaveInterval or DEFAULT_AUTOSAVE
	self.maxRetries = config.maxRetries or DEFAULT_RETRIES
	self.debug = config.Debug or false

	self.profiles = {}
	self.loading = {}
	self.running = true

	self:_startAutoSave()

	game:BindToClose(function()
		self:Shutdown()
	end)

	return self
end

function DataStoreClass:_log(...)
	if self.debug then
		print("[DataStore]", ...)
	end
end

function DataStoreClass:_key(userId: number): string
	return `Player_{userId}`
end

function DataStoreClass:_clone(tbl)
	local copy = {}

	for key, value in pairs(tbl) do
		if typeof(value) == "table" then
			copy[key] = self:_clone(value)
		else
			copy[key] = value
		end
	end

	return copy
end

function DataStoreClass:_reconcile(data, template)
	for key, value in pairs(template) do
		if data[key] == nil then
			data[key] = self:_clone(value)
		elseif typeof(value) == "table" and typeof(data[key]) == "table" then
			self:_reconcile(data[key], value)
		end
	end
end

function DataStoreClass:_retry(callback)
	local attempts = 0
	local success, result

	repeat
		attempts += 1
		success, result = pcall(callback)

		if success then
			return true, result
		end

		task.wait(math.min(2 ^ attempts, 8))
	until attempts >= self.maxRetries

	self:_log("Retry failed:", result)

	return false, result
end

function DataStoreClass:Load(userId: number)
	if self.profiles[userId] then
		return self.profiles[userId]
	end

	if self.loading[userId] then
		repeat task.wait() until not self.loading[userId]
		return self.profiles[userId]
	end

	self.loading[userId] = true

	local key = self:_key(userId)

	local success, data = self:_retry(function()
		return self.Store:GetAsync(key)
	end)

	local player = Players:GetPlayerByUserId(userId)

	if not player then
		self.loading[userId] = nil
		return nil
	end

	if not success then
		self.loading[userId] = nil
		player:Kick("Failed to load your data. Please rejoin.")
		return nil
	end

	if not data then
		data = self:_clone(self.template)
	else
		self:_reconcile(data, self.template)
	end

	local profile: Profile = {
		userId = userId,
		key = key,
		Data = data,
		loaded = true,
		dirty = false,
		sessionlock = true,
		lastsave = os.time(),
	}

	self.profiles[userId] = profile
	self.loading[userId] = nil

	self:_log("Loaded", userId)

	return profile
end

function DataStoreClass:Save(userId: number): boolean
	local profile: Profile = self.profiles[userId]

	if not profile then
		return false
	end

	local success = self:_retry(function()
		self.Store:UpdateAsync(profile.key, function()
			return profile.Data
		end)
	end)

	if success then
		profile.dirty = false
		profile.lastsave = os.time()
		self:_log("Saved", userId)
	end

	return success
end

function DataStoreClass:Update(userId: number, callback)
	local profile: Profile = self.profiles[userId]

	if not profile then
		return
	end

	callback(profile.Data)
	profile.dirty = true
end

function DataStoreClass:Get(userId: number)
	local profile: Profile = self.profiles[userId]
	return profile and profile.Data or nil
end

function DataStoreClass:Release(userId: number)
	local profile: Profile = self.profiles[userId]

	if not profile then
		return
	end

	self:Save(userId)
	self.profiles[userId] = nil

	self:_log("Released", userId)
end

function DataStoreClass:_startAutoSave()
	task.spawn(function()
		while self.running do
			task.wait(self.autosave)

			for userId, profile: Profile in pairs(self.profiles) do
				if profile.dirty then
					self:Save(userId)
					task.wait(0.15)
				end
			end
		end
	end)
end

function DataStoreClass:Shutdown()
	self.running = false

	for userId in pairs(self.profiles) do
		self:Release(userId)
	end
end

return DataStoreClass

Server Script


local DataService = require(script.DataService)
local PlayersData = require(script.PlayersData)
local Players = game:GetService('Players')

local DataStore = DataService.new({
	name = "PlayersDataStore",
	template = PlayersData,
	Debug = true,
	AutoSaveInterval = 60,
	maxRetries = 5,
})

Players.PlayerAdded:Connect(function(player)
	local profile = DataStore:Load(player.UserId)
	if not profile or not profile.Data then return end
	if player:FindFirstChild('leaderstats') then
		player.leaderstats:Destroy()
	end
	local ls = Instance.new('Folder')
	ls.Name = 'leaderstats'
	ls.Parent = player
	
	local Coins = Instance.new("NumberValue")
	Coins.Name = 'Coins'
	Coins.Value = profile.Data.Coins or 0
	Coins.Parent = ls
	
	local CoinsPlus = Instance.new("NumberValue")
	CoinsPlus.Name = 'CoinsPlus'
	CoinsPlus.Value = profile.Data.CoinsPlus or 1
	CoinsPlus.Parent = ls
end)

Players.PlayerRemoving:Connect(function(player)
	DataStore:Release(player.UserId)
end)
