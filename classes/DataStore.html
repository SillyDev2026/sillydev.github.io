DataStore System

Player saving system using DataStoreService, leaderstats creation, safe load/save logic, and automatic persistence handling.

Overview

Core features of this system

  • Loads player data safely on join
  • Creates leaderstats automatically
  • Saves data on player exit
  • Uses protected calls (pcall)
  • Supports multiple value types

Source Code

Lua implementation (copy-ready)


local DataStoreService = game:GetService('DataStoreService')
local PlayerStore = DataStoreService:GetDataStore("PlayerData")
local Players = game:GetService('Players')

local getData = function(player: Player)
	local success, result = pcall(function()
		return PlayerStore:GetAsync('Player_' .. player.UserId)
	end)

	if not success then
		player:Kick('Your data was corupted or failed to load')
	end

	if success and result then
		print(`Your data was resolved: {player.Name}`)
		return result
	end
end

local createFolders = function(player: Player, data)
	local ls = Instance.new('Folder')
	ls.Name = 'leaderstats'
	ls.Parent = player

	local Coins = Instance.new('NumberValue')
	Coins.Name = 'Coins'
	Coins.Value = data and data.Coins or 0
	Coins.Parent = ls
end

local loadData = function(player: Player)
	local data = getData(player) or {}
	createFolders(player, data)
end

local check = function(value): boolean
	return value:IsA("NumberValue")
		or value:IsA('StringValue')
		or value:IsA('BoolValue')
		or value:IsA('IntValue')
end

local saveTable = function(player: Player)
	local data = {}

	for _, folder in pairs(player:GetChildren()) do
		if folder:IsA('Folder') then
			for _, values in pairs(folder:GetChildren()) do
				if check(values) then
					data[values.Name] = values.Value
				end
			end
		end
	end

	return data
end

local saveData = function(player: Player)
	local data = saveTable(player)

	local success, save = pcall(function()
		return PlayerStore:SetAsync('Player_' .. player.UserId, data)
	end)

	if not success then
		warn(`Failed to save to [{player.Name}]`)
	else
		print(`Successfully saving: {player.Name}`)
	end
end

Players.PlayerAdded:Connect(loadData)
Players.PlayerRemoving:Connect(saveData)

game:BindToClose(function()
	task.wait(2)
end)
