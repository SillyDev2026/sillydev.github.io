What this system does.

Full Source Code

Lua formatted, copy-ready script.

local DataStoreService = game:GetService("DataStoreService")
local PlayerStore = DataStoreService:GetDataStore("PlayerData")
local Players = game:GetService("Players")

local function getData(player: Player)
	local success, result = pcall(function()
		return PlayerStore:GetAsync("Player_" .. player.UserId)
	end)

	if not success then
		player:Kick("Your data was corrupted or failed to load")
		return nil
	end

	if success and result then
		print(`Data loaded for {player.Name}`)
	end

	return result
end

local function createFolders(player: Player, data)
	local leaderstats = Instance.new("Folder")
	leaderstats.Name = "leaderstats"
	leaderstats.Parent = player

	local coins = Instance.new("IntValue")
	coins.Name = "Coins"
	coins.Value = data and data.Coins or 0
	coins.Parent = leaderstats
end

local function loadData(player: Player)
	local data = getData(player) or {}
	createFolders(player, data)
end

local function isValidValue(value): boolean
	return value:IsA("NumberValue")
		or value:IsA("IntValue")
		or value:IsA("BoolValue")
		or value:IsA("StringValue")
end

local function saveTable(player: Player)
	local data = {}

	for _, folder in pairs(player:GetChildren()) do
		if folder:IsA("Folder") then
			for _, value in pairs(folder:GetChildren()) do
				if isValidValue(value) then
					data[value.Name] = value.Value
				end
			end
		end
	end

	return data
end

local function saveData(player: Player)
	local data = saveTable(player)

	local success, err = pcall(function()
		PlayerStore:SetAsync("Player_" .. player.UserId, data)
	end)

	if not success then
		warn(`Failed to save data for {player.Name}`)
	else
		print(`Saved data for {player.Name}`)
	end
end

Players.PlayerAdded:Connect(loadData)
Players.PlayerRemoving:Connect(saveData)

game:BindToClose(function()
	for _, player in pairs(Players:GetPlayers()) do
		saveData(player)
	end
	task.wait(2)
end)

Notes

Important behavior details.

  • Data only saves in published games
  • Studio may cache old data
  • Avoid spam saving every frame
  • Always use pcall with DataStores
