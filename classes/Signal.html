Signal System

Fast custom event system with Connect, Once, Wait, Fire, DisconnectAll, and Destroy support.

Back View Source

Overview

Core features of this module

  • Create lightweight custom events
  • Connect unlimited listeners
  • Use Once for one-time callbacks
  • Yield with Wait()
  • Disconnect all listeners instantly
  • Destroy when finished

Setup Guide

How to install this system

  1. Create a ModuleScript named Signal
  2. Paste the source code below
  3. Require it in your scripts
  4. Create signals with Signal.new()
  5. Connect and Fire events

Example Usage

Quick example


local Signal = require(path.to.Signal)

local Event = Signal.new()

Event:Connect(function(playerName)
	print(playerName .. " joined")
end)

Event:Fire("Builderman")

Source Code

Lua implementation (copy-ready)


local Signal = {}
Signal.__index = Signal

local Connection = {}
Connection.__index = Connection

function Connection:Disconnect()
	if not self._node then
		return
	end

	self._node.Connected = false
	self._node.Callback = nil
	self._node = nil
end

function Connection:Connected()
	return self._node ~= nil
end

function Signal.new()
	return setmetatable({
		Head = nil,
		Destroyed = false
	}, Signal)
end

function Signal:Connect(fn)
	assert(type(fn) == "function", "Expected function")

	local node = {
		Callback = fn,
		Next = self.Head,
		Connected = true
	}

	self.Head = node

	return setmetatable({
		_node = node
	}, Connection)
end

function Signal:Once(fn)
	local connection

	connection = self:Connect(function(...)
		connection:Disconnect()
		fn(...)
	end)

	return connection
end

function Signal:Wait()
	local thread = coroutine.running()
	local connection

	connection = self:Connect(function(...)
		connection:Disconnect()
		task.spawn(thread, ...)
	end)

	return coroutine.yield()
end

function Signal:Fire(...)
	local node = self.Head

	while node do
		if node.Connected and node.Callback then
			task.spawn(node.Callback, ...)
		end

		node = node.Next
	end
end

function Signal:DisconnectAll()
	local node = self.Head

	while node do
		node.Connected = false
		node.Callback = nil
		node = node.Next
	end

	self.Head = nil
end

function Signal:Destroy()
	self:DisconnectAll()
	self.Destroyed = true
end

return Signal
